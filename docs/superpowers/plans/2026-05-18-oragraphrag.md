# OraGraphRAG Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build OraGraphRAG — a Python CLI + library that ingests a folder of documents into an Oracle AI Database 23ai property graph and answers questions using query-conditioned dynamic edge reweighting over a bounded subgraph, with a benchmark harness against naive RAG / GraphRAG / LightRAG on a 200-question Oracle 23ai docs Q&A suite.

**Architecture:** Two-stage retrieve-then-reweight (Approach A in the spec). Vector seed search via Oracle 23ai HNSW indexes → bounded PGQL subgraph extraction → in-Python edge reweighting via sigmoid-squashed projection of the query embedding onto five ontology-axis vectors (causal, taxonomic, temporal, definitional, exemplification) → personalized PageRank over the reweighted subgraph → JSON Duality View → LLM answer with proposition-ID citations. LLM layer is a thin adapter over OCI Generative AI (Grok 4.3, default) and Ollama (Qwen3.5-35B-A3B, local).

**Tech Stack:** Python 3.12, `oracledb`, `igraph`, `numpy`, `typer`, `httpx`, `oci` SDK, `pytest`, Oracle Database 23ai Free (Docker), OCI Generative AI service. Repo conventions follow `ragcli`: `pyproject.toml`, `requirements.txt`, `config.yaml.example`, `docker-compose.yml`, `Dockerfile`, `install.sh`.

**Spec:** `docs/superpowers/specs/2026-05-18-oragraphrag-design.md`

```python
"""Configuration loading: defaults < config.yaml < env vars."""

from __future__ import annotations

from pathlib import Path
from typing import Literal, Optional

import yaml
from pydantic import BaseModel, Field
from pydantic_settings import BaseSettings, SettingsConfigDict


class OciGrokConfig(BaseModel):
    compartment_ocid: str = ""
    endpoint_id: str = ""
    model: str = "grok-4-3"
    region: str = "us-chicago-1"


class OllamaConfig(BaseModel):
    base_url: str = "http://localhost:11434"
    model: str = "qwen3.5:35b-a3b"


class LlmConfig(BaseModel):
    provider: Literal["oci_grok", "ollama", "openai_compat"] = "oci_grok"
    oci_grok: OciGrokConfig = Field(default_factory=OciGrokConfig)
    ollama: OllamaConfig = Field(default_factory=OllamaConfig)
    fallback_on_outage: bool = False
    request_timeout_s: float = 60.0
    max_retries: int = 5


class OracleEmbedConfig(BaseModel):
    model: str = "ALL_MINILM_L12_V2"


class EmbeddingsConfig(BaseModel):
    provider: Literal["oracle", "ollama", "sentence_transformers"] = "oracle"
    oracle: OracleEmbedConfig = Field(default_factory=OracleEmbedConfig)
    dim: int = 384


class OracleConfig(BaseModel):
    username: str = "ORAGRAPH"
    password: str = "Welcome12345*"
    dsn: str = "localhost:1521/FREEPDB1"
    pool_min: int = 1
    pool_max: int = 8


class AmplitudeConfig(BaseModel):
    alpha: float = 8.0
    beta: float = 0.0
    per_axis_overrides: dict[str, float] = Field(default_factory=dict)


class PageRankConfig(BaseModel):
    damping: float = 0.85
    top_m_entities: int = 20


class RetrievalConfig(BaseModel):
    seed_k_entities: int = 8
    seed_k_propositions: int = 16
    max_subgraph_nodes: int = 256
    max_subgraph_edges: int = 2048
    amplitude: AmplitudeConfig = Field(default_factory=AmplitudeConfig)
    pagerank: PageRankConfig = Field(default_factory=PageRankConfig)


class AnswerConfig(BaseModel):
    token_budget: int = 4000


class IngestConfig(BaseModel):
    span_max_tokens: int = 1200
    section_overlap_tokens: int = 100
    extract_concurrency: int = 8
    canonicalize_threshold: float = 0.92
    hnsw_refresh_after_inserts: int = 1000


class Config(BaseSettings):
    model_config = SettingsConfigDict(
        env_prefix="OGR__",
        env_nested_delimiter="__",
        case_sensitive=False,
    )

    llm: LlmConfig = Field(default_factory=LlmConfig)
    embeddings: EmbeddingsConfig = Field(default_factory=EmbeddingsConfig)
    oracle: OracleConfig = Field(default_factory=OracleConfig)
    retrieval: RetrievalConfig = Field(default_factory=RetrievalConfig)
    answer: AnswerConfig = Field(default_factory=AnswerConfig)
    ingest: IngestConfig = Field(default_factory=IngestConfig)

    @classmethod
    def from_yaml(cls, path: str | Path) -> "Config":
        data = yaml.safe_load(Path(path).read_text()) or {}
        return cls(**data)
```

```python
"""LLM adapter: oci_grok | ollama | openai_compat backends, plus embedding pass-through."""

from __future__ import annotations

import json
from typing import Any, Optional, Protocol

import httpx
from tenacity import (
    AsyncRetrying,
    retry_if_exception_type,
    stop_after_attempt,
    wait_exponential,
)

from oragraphrag.config import Config, LlmConfig, OciGrokConfig, OllamaConfig


class LLMError(RuntimeError):
    pass


class _HttpClient(Protocol):
    async def post(self, url: str, json: Any = None, timeout: float | None = None): ...


class OllamaBackend:
    def __init__(self, cfg: OllamaConfig, http: _HttpClient | None = None):
        self.cfg = cfg
        self._http = http or httpx.AsyncClient()

    async def complete(
        self,
        prompt: str,
        *,
        schema: Optional[dict] = None,
        temperature: float = 0.0,
    ) -> str | dict:
        payload = {
            "model": self.cfg.model,
            "messages": [{"role": "user", "content": prompt}],
            "options": {"temperature": temperature},
            "format": "json" if schema else None,
            "stream": False,
        }
        payload = {k: v for k, v in payload.items() if v is not None}
        url = f"{self.cfg.base_url.rstrip('/')}/api/chat"
        resp = await self._http.post(url, json=payload, timeout=None)
        resp.raise_for_status()
        content = resp.json()["message"]["content"]
        if schema:
            try:
                return json.loads(content)
            except json.JSONDecodeError as e:
                raise LLMError(f"ollama returned non-JSON when schema requested: {e}") from e
        return content


class OciGrokBackend:
    """OCI Generative AI Grok 4.3 backend.

    Uses the oci.generative_ai_inference SDK. Auth via ~/.oci/config profile
    or instance principals (same pattern as the Oracle AI Developer Hub
    'Choose Your Path' apps).
    """

    def __init__(self, cfg: OciGrokConfig):
        self.cfg = cfg
        self._client = None  # lazy

    def _client_init(self):
        if self._client is None:
            import oci
            from oci.generative_ai_inference import GenerativeAiInferenceClient

            try:
                signer = oci.auth.signers.InstancePrincipalsSecurityTokenSigner()
                self._client = GenerativeAiInferenceClient(config={}, signer=signer)
            except Exception:
                cfg_file = oci.config.from_file()
                self._client = GenerativeAiInferenceClient(config=cfg_file)
        return self._client

    async def complete(
        self,
        prompt: str,
        *,
        schema: Optional[dict] = None,
        temperature: float = 0.0,
    ) -> str | dict:
        import asyncio
        from oci.generative_ai_inference.models import (
            ChatDetails,
            GenericChatRequest,
            Message,
            TextContent,
            OnDemandServingMode,
        )

        client = self._client_init()
        msg = Message(role="USER", content=[TextContent(text=prompt)])
        req = GenericChatRequest(
            messages=[msg],
            temperature=temperature,
            max_tokens=4096,
        )
        details = ChatDetails(
            compartment_id=self.cfg.compartment_ocid,
            serving_mode=OnDemandServingMode(model_id=self.cfg.model),
            chat_request=req,
        )
        resp = await asyncio.to_thread(client.chat, details)
        text = resp.data.chat_response.choices[0].message.content[0].text
        if schema:
            try:
                return json.loads(text)
            except json.JSONDecodeError as e:
                raise LLMError(f"oci_grok returned non-JSON when schema requested: {e}") from e
        return text


class LLM:
    def __init__(self, cfg: Config, http: _HttpClient | None = None):
        self.cfg = cfg
        self._http = http
        self._primary = self._build_primary()
        self._fallback = self._build_fallback() if cfg.llm.fallback_on_outage else None

    def _build_primary(self):
        return self._build(self.cfg.llm.provider)

    def _build_fallback(self):
        return self._build("ollama" if self.cfg.llm.provider != "ollama" else "oci_grok")

    def _build(self, provider: str):
        if provider == "ollama":
            return OllamaBackend(self.cfg.llm.ollama, http=self._http)
        if provider == "oci_grok":
            return OciGrokBackend(self.cfg.llm.oci_grok)
        raise LLMError(f"unsupported provider: {provider}")

    async def complete(self, prompt: str, *, schema: Optional[dict] = None, temperature: float = 0.0) -> str | dict:
        async for attempt in AsyncRetrying(
            stop=stop_after_attempt(self.cfg.llm.max_retries),
            wait=wait_exponential(multiplier=0.5, min=0.5, max=10.0),
            retry=retry_if_exception_type((httpx.HTTPError, LLMError)),
            reraise=True,
        ):
            with attempt:
                return await self._primary.complete(prompt, schema=schema, temperature=temperature)
        if self._fallback is not None:
            return await self._fallback.complete(prompt, schema=schema, temperature=temperature)
        raise LLMError("primary backend exhausted retries and no fallback configured")
```

```python
"""Embedding adapter. Wraps Oracle 23ai VECTOR_EMBEDDING, Ollama, or local ST."""

from __future__ import annotations

from typing import Iterable, Protocol

import numpy as np

from oragraphrag.axes import AXIS_DESCRIPTIONS, ONTOLOGY_AXIS_NAMES
from oragraphrag.config import Config


class _EmbedBackend(Protocol):
    dim: int
    async def embed(self, texts: list[str]) -> np.ndarray: ...


class Embedder:
    def __init__(self, cfg: Config, backend: _EmbedBackend):
        self.cfg = cfg
        self._backend = backend
        if backend.dim != cfg.embeddings.dim:
            raise ValueError(
                f"backend dim {backend.dim} != config dim {cfg.embeddings.dim}; "
                f"run `oragraphrag init-db --rebuild`"
            )

    async def embed(self, texts: Iterable[str], *, normalize: bool = True) -> np.ndarray:
        texts = list(texts)
        if not texts:
            return np.empty((0, self.cfg.embeddings.dim), dtype=np.float32)
        out = await self._backend.embed(texts)
        if normalize:
            out = self._l2_normalize(out)
        return out

    @staticmethod
    def _l2_normalize(a: np.ndarray) -> np.ndarray:
        n = np.linalg.norm(a, axis=1, keepdims=True)
        n = np.where(n == 0, 1.0, n)
        return a / n


async def build_axis_vectors(embedder) -> dict[str, np.ndarray]:
    """Embed the canonical descriptions for each ontology axis once.

    Returns a dict {axis_name: np.ndarray(dim,)}.
    """
    names = list(ONTOLOGY_AXIS_NAMES)
    descs = [AXIS_DESCRIPTIONS[n] for n in names]
    mat = await embedder.embed(descs)
    return {name: mat[i] for i, name in enumerate(names)}
```

```python
"""Oracle 23ai property-graph + vector store. The only module that talks to the DB."""

from __future__ import annotations

import json
from importlib.resources import files
from pathlib import Path
from typing import Iterable, Optional

import oracledb

from oragraphrag.config import Config


class GraphStore:
    def __init__(self, cfg: Config):
        self.cfg = cfg
        self._pool: Optional[oracledb.ConnectionPool] = None

    def connect(self) -> None:
        if self._pool is None:
            self._pool = oracledb.create_pool(
                user=self.cfg.oracle.username,
                password=self.cfg.oracle.password,
                dsn=self.cfg.oracle.dsn,
                min=self.cfg.oracle.pool_min,
                max=self.cfg.oracle.pool_max,
                increment=1,
            )

    def close(self) -> None:
        if self._pool is not None:
            self._pool.close()
            self._pool = None

    def _conn(self):
        assert self._pool is not None, "call connect() first"
        return self._pool.acquire()

    # ---------- DDL ----------

    def init_db(self, *, rebuild: bool, axis_vectors: dict[str, list[float]]) -> None:
        dim = self.cfg.embeddings.dim
        schema_sql = (files("oragraphrag.sql").joinpath("schema.sql").read_text()
                      .replace(":dim", str(dim)))
        with self._conn() as c, c.cursor() as cur:
            if rebuild:
                for stmt in (
                    "DROP PROPERTY GRAPH oragraph",
                    "DROP TABLE Mentions PURGE",
                    "DROP TABLE Rel PURGE",
                    "DROP TABLE Proposition PURGE",
                    "DROP TABLE Entity PURGE",
                    "DROP TABLE Ontology_Axis PURGE",
                    "DROP TABLE Ingest_Ledger PURGE",
                ):
                    try:
                        cur.execute(stmt)
                    except oracledb.DatabaseError:
                        pass
            for stmt in self._split_sql(schema_sql):
                cur.execute(stmt)
            for name, vec in axis_vectors.items():
                cur.execute(
                    "INSERT INTO Ontology_Axis (name, description, axis_embedding) VALUES (:n, :d, :v)",
                    n=name, d=name, v=vec,
                )
            c.commit()

    @staticmethod
    def _split_sql(text: str) -> list[str]:
        out, buf, in_block = [], [], False
        for line in text.splitlines():
            if line.strip().upper().startswith("CREATE PROPERTY GRAPH"):
                in_block = True
            buf.append(line)
            stripped = line.rstrip()
            if stripped.endswith(";") and not in_block:
                out.append("\n".join(buf).rstrip(";\n").strip())
                buf = []
            elif in_block and stripped.endswith(";"):
                out.append("\n".join(buf).rstrip(";\n").strip())
                buf = []
                in_block = False
        if buf and "\n".join(buf).strip():
            out.append("\n".join(buf).strip())
        return [s for s in out if s]

    def list_tables(self) -> list[str]:
        with self._conn() as c, c.cursor() as cur:
            cur.execute("SELECT table_name FROM user_tables")
            return [r[0] for r in cur.fetchall()]

    def list_property_graphs(self) -> list[str]:
        with self._conn() as c, c.cursor() as cur:
            cur.execute("SELECT graph_name FROM user_property_graphs")
            return [r[0] for r in cur.fetchall()]

    # ---------- Entity ops ----------

    def upsert_entity(self, *, name: str, kind: str, embedding: list[float]) -> bytes:
        with self._conn() as c, c.cursor() as cur:
            cur.execute("SELECT id FROM Entity WHERE LOWER(name) = LOWER(:n)", n=name)
            row = cur.fetchone()
            if row:
                eid = row[0]
                cur.execute(
                    "UPDATE Entity SET mention_count = mention_count + 1, embedding = :v WHERE id = :id",
                    v=embedding, id=eid,
                )
            else:
                out_id = cur.var(oracledb.DB_TYPE_RAW)
                cur.execute(
                    "INSERT INTO Entity (name, kind, embedding) VALUES (:n, :k, :v) RETURNING id INTO :id",
                    n=name, k=kind, v=embedding, id=out_id,
                )
                eid = out_id.getvalue()[0]
            c.commit()
            return eid

    def upsert_proposition(
        self, *, text: str, source_doc: str, source_span: str, embedding: list[float]
    ) -> bytes:
        with self._conn() as c, c.cursor() as cur:
            out_id = cur.var(oracledb.DB_TYPE_RAW)
            cur.execute(
                "INSERT INTO Proposition (text, source_doc, source_span, embedding) "
                "VALUES (:t, :d, :s, :v) RETURNING id INTO :id",
                t=text, d=source_doc, s=source_span, v=embedding, id=out_id,
            )
            c.commit()
            return out_id.getvalue()[0]

    def upsert_rel(
        self,
        src_id: bytes,
        dst_id: bytes,
        *,
        predicate: str,
        ontology_axis: str,
        base_weight: float,
        support_prop_id: bytes,
    ) -> None:
        with self._conn() as c, c.cursor() as cur:
            cur.execute(
                """
                MERGE INTO Rel r
                USING (SELECT :src AS src_id, :dst AS dst_id, :pred AS predicate FROM dual) k
                  ON (r.src_id = k.src_id AND r.dst_id = k.dst_id AND r.predicate = k.predicate)
                WHEN MATCHED THEN UPDATE SET
                  base_weight = (r.base_weight + :bw) / 2,
                  last_seen_at = SYSTIMESTAMP,
                  support_propositions = JSON_ARRAY_APPEND(NVL(r.support_propositions, JSON('[]')), '$', :sp_str),
                  support_axis_counts = JSON_MERGEPATCH(
                    NVL(r.support_axis_counts, JSON('{}')),
                    JSON_OBJECT(:axis VALUE
                      NVL(JSON_VALUE(r.support_axis_counts, '$.' || :axis), 0) + 1)
                  ),
                  ontology_axis = (
                    SELECT j.k FROM JSON_TABLE(
                      r.support_axis_counts, '$.*' COLUMNS (k PATH '$', v NUMBER PATH '$')
                    ) j ORDER BY j.v DESC FETCH FIRST 1 ROWS ONLY)
                WHEN NOT MATCHED THEN INSERT (src_id, dst_id, predicate, ontology_axis, base_weight,
                                              support_propositions, support_axis_counts)
                VALUES (:src, :dst, :pred, :axis, :bw,
                        JSON_ARRAY(:sp_str), JSON_OBJECT(:axis VALUE 1))
                """,
                src=src_id, dst=dst_id, pred=predicate, axis=ontology_axis,
                bw=base_weight, sp_str=support_prop_id.hex(),
            )
            c.commit()

    # ---------- Reads ----------

    def vector_search_entities(self, *, query_vec: list[float], k: int) -> list[dict]:
        with self._conn() as c, c.cursor() as cur:
            cur.execute(
                """
                SELECT id, name, VECTOR_DISTANCE(embedding, :q, COSINE) AS d
                FROM Entity ORDER BY d FETCH APPROX FIRST :k ROWS ONLY
                """,
                q=query_vec, k=k,
            )
            return [{"id": r[0], "name": r[1], "distance": float(r[2])} for r in cur.fetchall()]

    def vector_search_propositions(self, *, query_vec: list[float], k: int) -> list[dict]:
        with self._conn() as c, c.cursor() as cur:
            cur.execute(
                """
                SELECT id, source_doc, source_span, VECTOR_DISTANCE(embedding, :q, COSINE) AS d
                FROM Proposition ORDER BY d FETCH APPROX FIRST :k ROWS ONLY
                """,
                q=query_vec, k=k,
            )
            return [
                {"id": r[0], "source_doc": r[1], "source_span": r[2], "distance": float(r[3])}
                for r in cur.fetchall()
            ]

    def pgql_subgraph(self, *, seed_ids: list[bytes], max_edges: int) -> list[dict]:
        if not seed_ids:
            return []
        with self._conn() as c, c.cursor() as cur:
            cur.execute(
                """
                SELECT e1_id AS src, e2_id AS dst, predicate, ontology_axis, base_weight,
                       support_propositions
                FROM GRAPH_TABLE (oragraph
                  MATCH (e1) -[r IS REL]-> (e2)
                  WHERE ID(e1) IN :seeds OR ID(e2) IN :seeds
                  COLUMNS (ID(e1) AS e1_id, ID(e2) AS e2_id,
                           r.predicate AS predicate, r.ontology_axis AS ontology_axis,
                           r.base_weight AS base_weight,
                           r.support_propositions AS support_propositions)
                )
                FETCH FIRST :max_edges ROWS ONLY
                """,
                seeds=seed_ids, max_edges=max_edges,
            )
            return [
                {
                    "src": r[0], "dst": r[1], "predicate": r[2],
                    "ontology_axis": r[3], "base_weight": float(r[4]),
                    "support_propositions": json.loads(r[5].read()) if r[5] else [],
                }
                for r in cur.fetchall()
            ]

    def fetch_propositions(self, ids: Iterable[bytes]) -> list[dict]:
        ids = list(ids)
        if not ids:
            return []
        with self._conn() as c, c.cursor() as cur:
            cur.execute(
                "SELECT id, text, source_doc, source_span FROM Proposition WHERE id IN :ids",
                ids=ids,
            )
            return [
                {"id": r[0], "text": r[1].read() if r[1] else "",
                 "source_doc": r[2], "source_span": r[3]}
                for r in cur.fetchall()
            ]
```

```python
"""Walk a folder, parse files by type, emit normalized Spans, buffer into ~token-bounded chunks."""

from __future__ import annotations

import hashlib
import re
from dataclasses import dataclass, field
from pathlib import Path
from typing import Iterator

from markdown_it import MarkdownIt
from pypdf import PdfReader

_TOKEN_APPROX = 4  # chars per token, rough


@dataclass(slots=True)
class Span:
    doc_id: str
    section_path: str
    text: str
    span_id: str = ""

    @property
    def hash(self) -> str:
        h = hashlib.sha256()
        h.update(self.doc_id.encode())
        h.update(b"\x00")
        h.update(self.section_path.encode())
        h.update(b"\x00")
        h.update(self.text.encode())
        return h.hexdigest()


@dataclass(slots=True)
class Buffer:
    doc_id: str
    section_path: str
    text: str
    span_hashes: list[str] = field(default_factory=list)

    @property
    def approx_tokens(self) -> int:
        return max(1, len(self.text) // _TOKEN_APPROX)


_SUPPORTED_EXTS = {".md", ".markdown", ".txt", ".pdf"}


def walk_folder(root: Path) -> Iterator[Span]:
    root = Path(root)
    for p in sorted(root.rglob("*")):
        if not p.is_file() or p.suffix.lower() not in _SUPPORTED_EXTS:
            continue
        rel = p.relative_to(root).as_posix()
        if p.suffix.lower() in {".md", ".markdown"}:
            yield from _parse_markdown(p, rel)
        elif p.suffix.lower() == ".txt":
            yield Span(doc_id=rel, section_path="", text=p.read_text(errors="replace"))
        elif p.suffix.lower() == ".pdf":
            yield from _parse_pdf(p, rel)


def _parse_markdown(path: Path, rel: str) -> Iterator[Span]:
    md = MarkdownIt()
    tokens = md.parse(path.read_text(errors="replace"))
    section_stack: list[str] = []
    buf: list[str] = []

    def emit():
        if not buf:
            return None
        text = "\n\n".join(buf).strip()
        buf.clear()
        if not text:
            return None
        return Span(doc_id=rel, section_path=" / ".join(section_stack), text=text)

    i = 0
    while i < len(tokens):
        tok = tokens[i]
        if tok.type == "heading_open":
            span = emit()
            if span:
                yield span
            inline = tokens[i + 1]
            level = int(tok.tag[1]) if tok.tag.startswith("h") else 1
            section_stack[level - 1 :] = [inline.content]
            i += 3
            continue
        if tok.type == "inline":
            buf.append(tok.content)
        i += 1
    span = emit()
    if span:
        yield span


def _parse_pdf(path: Path, rel: str) -> Iterator[Span]:
    reader = PdfReader(str(path))
    for n, page in enumerate(reader.pages):
        text = page.extract_text() or ""
        if text.strip():
            yield Span(doc_id=rel, section_path=f"page-{n+1}", text=text.strip())


def buffer_spans(
    spans: Iterator[Span], *, max_tokens: int, overlap_tokens: int
) -> Iterator[Buffer]:
    cur_doc: str | None = None
    cur_section: str | None = None
    cur_text: list[str] = []
    cur_hashes: list[str] = []
    cur_tokens = 0

    def flush() -> Buffer | None:
        nonlocal cur_text, cur_hashes, cur_tokens
        if not cur_text:
            return None
        b = Buffer(doc_id=cur_doc or "", section_path=cur_section or "",
                   text="\n\n".join(cur_text).strip(),
                   span_hashes=list(cur_hashes))
        # carry overlap text into the next buffer
        if overlap_tokens > 0 and cur_text:
            tail = cur_text[-1]
            tail_chars = overlap_tokens * _TOKEN_APPROX
            cur_text = [tail[-tail_chars:]] if len(tail) > tail_chars else [tail]
            cur_hashes = []
            cur_tokens = len(cur_text[0]) // _TOKEN_APPROX
        else:
            cur_text = []
            cur_hashes = []
            cur_tokens = 0
        return b

    for span in spans:
        section_changed = (span.section_path != cur_section) or (span.doc_id != cur_doc)
        if section_changed and cur_text:
            b = flush()
            if b:
                yield b
            cur_text = []
            cur_hashes = []
            cur_tokens = 0
        cur_doc = span.doc_id
        cur_section = span.section_path

        span_tokens = max(1, len(span.text) // _TOKEN_APPROX)
        if cur_tokens + span_tokens > max_tokens and cur_text:
            b = flush()
            if b:
                yield b
        cur_text.append(span.text)
        cur_hashes.append(span.hash)
        cur_tokens += span_tokens

    b = flush()
    if b:
        yield b
```

```python
"""LLM proposition + triple extraction with strict JSON validation."""

from __future__ import annotations

import json
from importlib.resources import files
from typing import Any

from jinja2 import Template

from oragraphrag.axes import ONTOLOGY_AXIS_NAMES


class ExtractionError(RuntimeError):
    pass


EXTRACT_SCHEMA: dict[str, Any] = {
    "type": "object",
    "required": ["propositions"],
    "properties": {
        "propositions": {
            "type": "array",
            "items": {
                "type": "object",
                "required": ["text", "triples"],
                "properties": {
                    "text": {"type": "string", "minLength": 1},
                    "triples": {
                        "type": "array",
                        "items": {
                            "type": "object",
                            "required": ["subject", "predicate", "object",
                                         "ontology_axis", "confidence"],
                            "properties": {
                                "subject": {"type": "string", "minLength": 1},
                                "predicate": {"type": "string", "minLength": 1},
                                "object": {"type": "string", "minLength": 1},
                                "ontology_axis": {"enum": list(ONTOLOGY_AXIS_NAMES)},
                                "confidence": {"type": "number", "minimum": 0, "maximum": 1},
                            },
                        },
                    },
                },
            },
        }
    },
}


def validate_payload(payload: Any) -> dict:
    if not isinstance(payload, dict) or "propositions" not in payload:
        raise ExtractionError("missing top-level 'propositions'")
    props = payload["propositions"]
    if not isinstance(props, list):
        raise ExtractionError("'propositions' must be a list")
    for p in props:
        if not isinstance(p, dict) or "text" not in p or "triples" not in p:
            raise ExtractionError("proposition missing 'text' or 'triples'")
        if not isinstance(p["text"], str) or not p["text"].strip():
            raise ExtractionError("proposition.text must be non-empty string")
        for t in p["triples"]:
            for k in ("subject", "predicate", "object", "ontology_axis", "confidence"):
                if k not in t:
                    raise ExtractionError(f"triple missing key: {k}")
            if t["ontology_axis"] not in ONTOLOGY_AXIS_NAMES:
                raise ExtractionError(f"bad axis: {t['ontology_axis']}")
            if not (0.0 <= float(t["confidence"]) <= 1.0):
                raise ExtractionError(f"bad confidence: {t['confidence']}")
    return payload


class Extractor:
    def __init__(self, llm, template_text: str | None = None):
        self._llm = llm
        text = template_text or files("oragraphrag.prompts").joinpath("extract.j2").read_text()
        self._tmpl = Template(text)

    async def extract(self, passage: str) -> dict:
        prompt = self._tmpl.render(passage=passage)
        try:
            payload = await self._llm.complete(prompt, schema=EXTRACT_SCHEMA, temperature=0.0)
            if isinstance(payload, str):
                payload = json.loads(payload)
            return validate_payload(payload)
        except (ExtractionError, json.JSONDecodeError, ValueError):
            repair = (
                "Your previous response was invalid JSON or violated the schema. "
                "Respond ONLY with valid JSON matching the schema above. PASSAGE:\n"
                f"\"\"\"\n{passage}\n\"\"\""
            )
            payload = await self._llm.complete(repair, schema=EXTRACT_SCHEMA, temperature=0.0)
            if isinstance(payload, str):
                payload = json.loads(payload)
            return validate_payload(payload)
```

```python
"""Orchestrates: buffers -> extract -> canonicalize entities -> embed -> graph upserts."""

from __future__ import annotations

import asyncio
import math
from typing import Iterable

import numpy as np

from oragraphrag.config import Config
from oragraphrag.ingest import Buffer


class IngestPipeline:
    def __init__(self, *, cfg: Config, graph, embedder, extractor):
        self.cfg = cfg
        self.graph = graph
        self.embedder = embedder
        self.extractor = extractor

    async def run(self, buffers: Iterable[Buffer]) -> dict:
        stats = {"buffers": 0, "skipped": 0, "propositions": 0, "rels": 0, "entities": 0}
        sem = asyncio.Semaphore(self.cfg.ingest.extract_concurrency)

        async def process(buf: Buffer):
            stats["buffers"] += 1
            # Skip if every span in the buffer is already ledgered.
            if buf.span_hashes and all(self.graph.ledger_has(h) for h in buf.span_hashes):
                stats["skipped"] += 1
                return
            async with sem:
                payload = await self.extractor.extract(buf.text)

            prop_texts = [p["text"] for p in payload["propositions"]]
            if not prop_texts:
                for h in buf.span_hashes:
                    self.graph.ledger_add(h, doc_id=buf.doc_id, section_path=buf.section_path)
                return
            prop_embs = await self.embedder.embed(prop_texts)

            entity_set: list[str] = []
            seen: set[str] = set()
            for p in payload["propositions"]:
                for t in p["triples"]:
                    for k in (t["subject"], t["object"]):
                        if k not in seen:
                            seen.add(k)
                            entity_set.append(k)
            entity_embs = await self.embedder.embed(entity_set) if entity_set else np.empty((0, self.cfg.embeddings.dim))

            entity_ids: dict[str, bytes] = {}
            for name, vec in zip(entity_set, entity_embs):
                eid = self.graph.upsert_entity(name=name, kind="concept", embedding=vec.tolist())
                entity_ids[name] = eid
                stats["entities"] += 1

            for prop, vec in zip(payload["propositions"], prop_embs):
                pid = self.graph.upsert_proposition(
                    text=prop["text"], source_doc=buf.doc_id,
                    source_span=buf.section_path, embedding=vec.tolist(),
                )
                stats["propositions"] += 1
                for t in prop["triples"]:
                    s_id = entity_ids[t["subject"]]
                    o_id = entity_ids[t["object"]]
                    base = 0.5 + 0.5 * float(t["confidence"])
                    self.graph.upsert_rel(
                        s_id, o_id,
                        predicate=t["predicate"],
                        ontology_axis=t["ontology_axis"],
                        base_weight=min(1.0, base),
                        support_prop_id=pid,
                    )
                    stats["rels"] += 1

            for h in buf.span_hashes:
                self.graph.ledger_add(h, doc_id=buf.doc_id, section_path=buf.section_path)

        await asyncio.gather(*(process(b) for b in buffers))
        return stats
```

```python
"""Pure-Python compute over a bounded subgraph: reweight + spreading activation + assembly."""

from __future__ import annotations

from typing import Iterable

import numpy as np
import igraph as ig

from oragraphrag.axes import ONTOLOGY_AXIS_NAMES


def compute_amplitudes(
    query_vec: np.ndarray,
    axis_vectors: dict[str, np.ndarray],
    *,
    alpha: float,
    beta: float,
) -> dict[str, float]:
    qn = query_vec / max(float(np.linalg.norm(query_vec)), 1e-12)
    out: dict[str, float] = {}
    for name in ONTOLOGY_AXIS_NAMES:
        av = axis_vectors[name]
        an = av / max(float(np.linalg.norm(av)), 1e-12)
        proj = float(np.dot(qn, an))
        out[name] = float(1.0 / (1.0 + np.exp(-(alpha * proj + beta))))
    return out


def reweight_edges(edges: list[dict], amplitudes: dict[str, float]) -> list[dict]:
    out = []
    for e in edges:
        amp = amplitudes.get(e["ontology_axis"], 0.5)
        new = dict(e)
        new["weight"] = float(e["base_weight"]) * amp
        out.append(new)
    return out


def spreading_activation(
    edges: list[dict],
    *,
    seed_ids: list[bytes],
    damping: float,
) -> dict[bytes, float]:
    node_ids: list[bytes] = []
    idx: dict[bytes, int] = {}
    def _idx(n: bytes) -> int:
        if n not in idx:
            idx[n] = len(node_ids)
            node_ids.append(n)
        return idx[n]
    for s in seed_ids:
        _idx(s)
    pairs = [(_idx(e["src"]), _idx(e["dst"])) for e in edges]
    weights = [float(e["weight"]) for e in edges]
    if not pairs:
        return {seed_ids[0]: 1.0} if seed_ids else {}
    g = ig.Graph(n=len(node_ids), edges=pairs, directed=True)
    g.es["weight"] = weights
    reset = [1.0 if n in set(seed_ids) else 0.0 for n in node_ids]
    s = sum(reset) or 1.0
    reset = [x / s for x in reset]
    scores = g.personalized_pagerank(reset=reset, weights="weight", damping=damping, implementation="prpack")
    return {node_ids[i]: float(scores[i]) for i in range(len(node_ids))}


def assemble_propositions(
    edges: list[dict],
    activations: dict[bytes, float],
    seed_sims: dict[bytes, float],
    *,
    top_m: int,
) -> list[bytes]:
    prop_score: dict[bytes, float] = {}
    for e in edges:
        act = max(activations.get(e["src"], 0.0), activations.get(e["dst"], 0.0))
        for ph in e.get("support_propositions", []):
            pid = bytes.fromhex(ph) if isinstance(ph, str) else ph
            seed = seed_sims.get(pid, 0.0)
            score = act * (0.5 + seed)
            if score > prop_score.get(pid, 0.0):
                prop_score[pid] = score
    return [p for p, _ in sorted(prop_score.items(), key=lambda kv: kv[1], reverse=True)[:top_m]]
```

```python
"""Render the answer prompt, call the LLM, map [P#] citations back to propositions."""

from __future__ import annotations

import re
from dataclasses import dataclass
from importlib.resources import files
from typing import Sequence

from jinja2 import Template


@dataclass(slots=True)
class Citation:
    proposition_id: bytes
    source_doc: str
    source_span: str


@dataclass(slots=True)
class AnswerResult:
    text: str
    citations: list[Citation]


_NO_INFO = "I don't have information about that in the indexed corpus."


class Answerer:
    def __init__(self, *, llm, token_budget: int, template_text: str | None = None):
        self._llm = llm
        self._budget = token_budget
        text = template_text or files("oragraphrag.prompts").joinpath("answer.j2").read_text()
        self._tmpl = Template(text)

    async def answer(self, *, question: str, propositions: Sequence[dict]) -> AnswerResult:
        if not propositions:
            return AnswerResult(text=_NO_INFO, citations=[])
        items = [
            {"id_hex": (p["id"].hex() if isinstance(p["id"], bytes) else str(p["id"])),
             "text": p["text"], "source_doc": p.get("source_doc", ""),
             "source_span": p.get("source_span", "")}
            for p in propositions
        ]
        prompt = self._tmpl.render(question=question, propositions=items)
        text = await self._llm.complete(prompt, temperature=0.0)
        if not isinstance(text, str):
            text = str(text)
        cits: list[Citation] = []
        for m in re.finditer(r"\[P(\d+)\]", text):
            n = int(m.group(1))
            if 1 <= n <= len(propositions):
                p = propositions[n - 1]
                cits.append(Citation(
                    proposition_id=p["id"] if isinstance(p["id"], bytes) else bytes.fromhex(p["id"]),
                    source_doc=p.get("source_doc", ""),
                    source_span=p.get("source_span", ""),
                ))
        return AnswerResult(text=text, citations=cits)
```

```python
"""End-to-end query: embed -> seeds -> subgraph -> reweight -> ppr -> assemble -> answer."""

from __future__ import annotations

import time
from dataclasses import dataclass

import numpy as np

from oragraphrag.answer import AnswerResult, Answerer
from oragraphrag.config import Config
from oragraphrag.retrieve import (
    assemble_propositions,
    compute_amplitudes,
    reweight_edges,
    spreading_activation,
)


@dataclass(slots=True)
class QueryResult:
    answer: AnswerResult
    amplitudes: dict[str, float]
    edges_used: list[dict]
    latency_ms: dict[str, float]


class QueryPipeline:
    def __init__(self, *, cfg: Config, graph, embedder, llm, axis_vectors: dict[str, np.ndarray]):
        self.cfg = cfg
        self.graph = graph
        self.embedder = embedder
        self.llm = llm
        self.axis_vectors = axis_vectors
        self._answerer = Answerer(llm=llm, token_budget=cfg.answer.token_budget)

    async def query(self, question: str) -> QueryResult:
        t = {}
        t0 = time.perf_counter()
        q = (await self.embedder.embed([question]))[0]
        t["embed_ms"] = (time.perf_counter() - t0) * 1000

        t0 = time.perf_counter()
        ent_seeds = self.graph.vector_search_entities(
            query_vec=q.tolist(), k=self.cfg.retrieval.seed_k_entities)
        prop_seeds = self.graph.vector_search_propositions(
            query_vec=q.tolist(), k=self.cfg.retrieval.seed_k_propositions)
        t["seed_ms"] = (time.perf_counter() - t0) * 1000

        seed_ids = [s["id"] for s in ent_seeds]
        seed_sims = {p["id"]: 1.0 - p["distance"] for p in prop_seeds}

        t0 = time.perf_counter()
        edges = self.graph.pgql_subgraph(
            seed_ids=seed_ids, max_edges=self.cfg.retrieval.max_subgraph_edges
        )
        t["subgraph_ms"] = (time.perf_counter() - t0) * 1000

        t0 = time.perf_counter()
        amps = compute_amplitudes(
            q, self.axis_vectors,
            alpha=self.cfg.retrieval.amplitude.alpha,
            beta=self.cfg.retrieval.amplitude.beta,
        )
        if all(a < 0.05 or a > 0.95 for a in amps.values()):
            # Degenerate amplitudes — fall back to base-weight PPR.
            amps = {k: 1.0 for k in amps}
        reweighted = reweight_edges(edges, amps)
        t["reweight_ms"] = (time.perf_counter() - t0) * 1000

        t0 = time.perf_counter()
        activations = spreading_activation(
            reweighted, seed_ids=seed_ids,
            damping=self.cfg.retrieval.pagerank.damping,
        )
        t["ppr_ms"] = (time.perf_counter() - t0) * 1000

        picked = assemble_propositions(
            reweighted, activations, seed_sims,
            top_m=self.cfg.retrieval.pagerank.top_m_entities,
        )
        props = self.graph.fetch_propositions(picked) if picked else []

        t0 = time.perf_counter()
        ans = await self._answerer.answer(question=question, propositions=props)
        t["answer_ms"] = (time.perf_counter() - t0) * 1000

        return QueryResult(answer=ans, amplitudes=amps, edges_used=reweighted, latency_ms=t)
```

```python
"""Concrete embedding backends: Oracle VECTOR_EMBEDDING, Ollama, sentence-transformers."""

from __future__ import annotations

from typing import Protocol

import httpx
import numpy as np

from oragraphrag.config import Config


class _HttpClient(Protocol):
    async def post(self, url, json=None, timeout=None): ...


class OllamaEmbedBackend:
    def __init__(self, cfg: Config, http: _HttpClient | None = None):
        self.cfg = cfg
        self.dim = cfg.embeddings.dim
        self._http = http or httpx.AsyncClient()

    async def embed(self, texts: list[str]) -> np.ndarray:
        out = np.zeros((len(texts), self.dim), dtype=np.float32)
        for i, t in enumerate(texts):
            resp = await self._http.post(
                f"{self.cfg.llm.ollama.base_url.rstrip('/')}/api/embeddings",
                json={"model": "nomic-embed-text", "prompt": t},
                timeout=None,
            )
            resp.raise_for_status()
            out[i] = np.array(resp.json()["embedding"], dtype=np.float32)
        return out


class OracleEmbedBackend:
    """Uses VECTOR_EMBEDDING(:text USING :model) via the connection pool on GraphStore."""

    def __init__(self, cfg: Config, graph):
        self.cfg = cfg
        self.dim = cfg.embeddings.dim
        self._graph = graph

    async def embed(self, texts: list[str]) -> np.ndarray:
        # VECTOR_EMBEDDING is synchronous; we keep it on the calling thread.
        import asyncio
        return await asyncio.to_thread(self._embed_sync, texts)

    def _embed_sync(self, texts: list[str]) -> np.ndarray:
        out = np.zeros((len(texts), self.dim), dtype=np.float32)
        with self._graph._conn() as c, c.cursor() as cur:
            for i, t in enumerate(texts):
                cur.execute(
                    "SELECT VECTOR_EMBEDDING(:m USING :t AS data) FROM dual",
                    m=self.cfg.embeddings.oracle.model, t=t,
                )
                row = cur.fetchone()
                out[i] = np.array(row[0], dtype=np.float32)
        return out


class SentenceTransformersBackend:
    def __init__(self, cfg: Config, model_name: str = "sentence-transformers/all-MiniLM-L12-v2"):
        from sentence_transformers import SentenceTransformer
        self.cfg = cfg
        self.dim = cfg.embeddings.dim
        self._m = SentenceTransformer(model_name)
        if self._m.get_sentence_embedding_dimension() != cfg.embeddings.dim:
            raise ValueError("model dim mismatch")

    async def embed(self, texts: list[str]) -> np.ndarray:
        import asyncio
        arr = await asyncio.to_thread(self._m.encode, texts, normalize_embeddings=False)
        return np.asarray(arr, dtype=np.float32)


def build_embed_backend(cfg: Config, graph):
    if cfg.embeddings.provider == "oracle":
        return OracleEmbedBackend(cfg, graph=graph)
    if cfg.embeddings.provider == "ollama":
        return OllamaEmbedBackend(cfg)
    if cfg.embeddings.provider == "sentence_transformers":
        return SentenceTransformersBackend(cfg)
    raise ValueError(f"unknown embeddings.provider: {cfg.embeddings.provider}")
```

```python
from __future__ import annotations

from pathlib import Path

from pyvis.network import Network


def render_subgraph_html(edges, *, activations: dict[bytes, float], out_path: Path) -> None:
    net = Network(directed=True, height="600px", width="100%", bgcolor="#111", font_color="#eee")
    seen = set()
    for e in edges:
        for n in (e["src"], e["dst"]):
            if n in seen:
                continue
            seen.add(n)
            act = activations.get(n, 0.0)
            net.add_node(n.hex(), label=n.hex()[:8], value=max(act, 0.01) * 100,
                         color={"background": "#5af", "border": "#fff"})
        net.add_edge(
            e["src"].hex(), e["dst"].hex(),
            value=max(e.get("weight", 0.0), 0.01) * 10,
            title=f"{e['predicate']} ({e['ontology_axis']})",
            label=e["ontology_axis"],
        )
    out_path = Path(out_path)
    out_path.parent.mkdir(parents=True, exist_ok=True)
    net.write_html(str(out_path), open_browser=False, notebook=False)
```

```python
"""Metrics: correctness (LLM-judge), citation P/R, recall@k."""

from __future__ import annotations

from importlib.resources import files
from typing import Sequence

from jinja2 import Template


_JUDGE_TMPL = Template(files("oragraphrag.prompts").joinpath("judge.j2").read_text())


async def _judge_call(question: str, gold: str, predicted: str, llm) -> int:
    prompt = _JUDGE_TMPL.render(question=question, gold=gold, predicted=predicted)
    raw = await llm.complete(prompt, temperature=0.0)
    try:
        return max(0, min(4, int(str(raw).strip().split()[0])))
    except (ValueError, IndexError):
        return 0


def score_correctness(*, question: str, gold: str, predicted: str, llm) -> int:
    import asyncio
    return asyncio.get_event_loop().run_until_complete(
        _judge_call(question, gold, predicted, llm)
    )


def citation_pr(*, gold: Sequence[str], pred: Sequence[str]) -> tuple[float, float]:
    g, p = set(gold), set(pred)
    if not p and not g:
        return 1.0, 1.0
    precision = len(g & p) / max(len(p), 1)
    recall = len(g & p) / max(len(g), 1)
    return precision, recall


def recall_at_k(*, gold: Sequence[str], retrieved: Sequence[str], k: int) -> float:
    g = set(gold)
    top = set(retrieved[:k])
    if not g:
        return 1.0
    return len(g & top) / len(g)
```

```python
"""Run a BenchSuite against one or more baseline systems."""

from __future__ import annotations

import json
import statistics
import time
from dataclasses import dataclass
from pathlib import Path
from typing import Iterable

from oragraphrag.bench.baselines import REGISTRY
from oragraphrag.bench.metrics import citation_pr, recall_at_k, _judge_call
from oragraphrag.config import Config
from oragraphrag.llm import LLM


@dataclass(slots=True)
class BenchSuite:
    items: list[dict]


def load_suite(path: str | Path) -> BenchSuite:
    return BenchSuite(items=[json.loads(line) for line in Path(path).read_text().splitlines() if line.strip()])


async def _run_one(system: str, question: str, cfg: Config) -> dict:
    runner = REGISTRY[system]
    t0 = time.perf_counter()
    out = await runner.run(question, cfg)
    out["latency_ms"] = (time.perf_counter() - t0) * 1000
    return out


async def run_suite(
    cfg: Config,
    *,
    suite: str,
    systems: Iterable[str],
    limit: int | None,
) -> dict:
    items = load_suite(suite).items
    if limit:
        items = items[:limit]
    llm = LLM(cfg)
    results: dict[str, dict] = {}
    for system in systems:
        per_q: list[dict] = []
        for it in items:
            out = await _run_one(system, it["question"], cfg)
            score = await _judge_call(it["question"], it["gold_answer"], out["answer"], llm)
            p, r = citation_pr(gold=it.get("gold_doc_ids", []), pred=out.get("citations", []))
            per_q.append({
                "id": it.get("id"),
                "hops": it.get("hops", 1),
                "correctness": score,
                "citation_precision": p,
                "citation_recall": r,
                "latency_ms": out["latency_ms"],
                "tokens": out.get("tokens", 0),
            })
        results[system] = {
            "n": len(per_q),
            "median_correctness": statistics.median(q["correctness"] for q in per_q) if per_q else 0,
            "median_latency_ms": statistics.median(q["latency_ms"] for q in per_q) if per_q else 0,
            "median_tokens": statistics.median(q["tokens"] for q in per_q) if per_q else 0,
            "median_citation_precision": statistics.median(q["citation_precision"] for q in per_q) if per_q else 0,
            "median_citation_recall": statistics.median(q["citation_recall"] for q in per_q) if per_q else 0,
            "per_question": per_q,
        }
    return {"suite": suite, "systems": results}
```
